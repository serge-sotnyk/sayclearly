# Stage 3 Guided Exercise Flow UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the landing placeholder with a usable setup and guided exercise flow that saves config, generates placeholder text through a new backend API, and advances the user to retelling readiness.

**Architecture:** Keep FastAPI and Jinja as the shell, add a focused `exercise` backend domain for `POST /api/generate-text`, and drive the browser flow with one plain TypeScript module plus a small pure state helper module. Commit the compiled frontend bundle under `src/sayclearly/static/dist/` so the local Python app can run after `uv sync` without requiring a runtime build step.

**Tech Stack:** Python 3.13+, FastAPI, Jinja2, Pydantic, pytest, Ruff, TypeScript compiler, Node built-in test runner

Create `src/sayclearly/exercise/service.py` with:

```python
from pathlib import Path

from sayclearly.exercise.models import ExerciseGenerationRequest, ExerciseGenerationResponse
from sayclearly.storage.files import load_config


class ExerciseService:
    def __init__(self, data_root: Path | None = None) -> None:
        self.data_root = data_root

    def generate_text(
        self,
        request: ExerciseGenerationRequest,
    ) -> ExerciseGenerationResponse:
        effective_topic = self._resolve_topic(request)
        subject = effective_topic or "everyday communication"
        text = " ".join(
            [
                f"This is a placeholder practice text about {subject}.",
                "It is meant to be read aloud slowly and clearly.",
                "Each sentence is short enough to keep the rhythm controlled.",
                "The wording stays simple so the speaker can focus on articulation.",
                "After reading, the speaker can briefly retell the same idea in their own words.",
                "Later stages will replace this placeholder with Gemini-generated content.",
            ]
        )

        return ExerciseGenerationResponse(
            text=text,
            language=request.language,
            analysis_language=request.analysis_language,
            topic_prompt=effective_topic,
        )

    def _resolve_topic(self, request: ExerciseGenerationRequest) -> str:
        topic = request.topic_prompt.strip()
        if topic:
            return topic
        if request.reuse_last_topic:
            return load_config(self.data_root).last_topic_prompt
        return ""
```

Create `src/sayclearly/exercise/api.py` with:

```python
from pathlib import Path

from fastapi import APIRouter, HTTPException

from sayclearly.exercise.models import ExerciseGenerationRequest, ExerciseGenerationResponse
from sayclearly.exercise.service import ExerciseService
from sayclearly.storage.files import StorageError


def build_exercise_router(data_root: Path | None = None) -> APIRouter:
    service = ExerciseService(data_root)
    router = APIRouter()

    @router.post("/api/generate-text", response_model=ExerciseGenerationResponse)
    def generate_text(payload: ExerciseGenerationRequest) -> ExerciseGenerationResponse:
        try:
            return service.generate_text(payload)
        except StorageError as exc:
            raise HTTPException(status_code=500, detail=str(exc)) from exc

    return router
```

```python
from pathlib import Path

from fastapi import FastAPI, Request
from fastapi.staticfiles import StaticFiles
from fastapi.templating import Jinja2Templates

from sayclearly.config.api import build_config_router
from sayclearly.exercise.api import build_exercise_router
from sayclearly.history.api import build_history_router
from sayclearly.web.errors import install_error_handlers

PACKAGE_ROOT = Path(__file__).resolve().parent
TEMPLATES_DIR = PACKAGE_ROOT / "templates"
STATIC_DIR = PACKAGE_ROOT / "static"


def create_app(data_root: Path | None = None) -> FastAPI:
    templates = Jinja2Templates(directory=str(TEMPLATES_DIR))
    app = FastAPI(
        title="SayClearly",
    )

    install_error_handlers(app)
    app.mount("/static", StaticFiles(directory=STATIC_DIR), name="static")
    app.include_router(build_config_router(data_root))
    app.include_router(build_history_router(data_root))
    app.include_router(build_exercise_router(data_root))

    @app.get("/")
    def home(request: Request):
        return templates.TemplateResponse(
            request=request,
            name="index.html",
            context={"page_title": "SayClearly"},
        )

    @app.get("/api/health")
    def health() -> dict[str, str]:
        return {"status": "ok"}

    return app
```

```typescript
export type FlowState =
  | "home"
  | "generating_text"
  | "step_1_slow"
  | "step_2_natural"
  | "step_3_retell_ready"
  | "error";

export interface PublicConfig {
  version: number;
  text_language: string;
  analysis_language: string;
  same_language_for_analysis: boolean;
  ui_language: string;
  last_topic_prompt: string;
  session_limit: number;
  keep_last_audio: boolean;
  gemini: {
    model: string;
    has_api_key: boolean;
    api_key_source: string;
  };
  langfuse: {
    host: string | null;
    enabled: boolean;
    has_public_key: boolean;
    has_secret_key: boolean;
    public_key_source: string;
    secret_key_source: string;
  };
}

export interface SettingsFormState {
  text_language: string;
  analysis_language: string;
  same_language_for_analysis: boolean;
  topic_prompt: string;
  reuse_last_topic: boolean;
  api_key: string;
}

export interface GeneratedExercise {
  text: string;
  language: string;
  analysis_language: string;
  topic_prompt: string;
}

export interface AppModel {
  state: FlowState;
  config: PublicConfig | null;
  settings: SettingsFormState;
  exercise: GeneratedExercise | null;
  errorMessage: string | null;
  statusMessage: string | null;
  settingsOpen: boolean;
}

export function createInitialAppModel(): AppModel {
  return {
    state: "home",
    config: null,
    settings: {
      text_language: "uk",
      analysis_language: "uk",
      same_language_for_analysis: true,
      topic_prompt: "",
      reuse_last_topic: false,
      api_key: "",
    },
    exercise: null,
    errorMessage: null,
    statusMessage: null,
    settingsOpen: false,
  };
}

export function syncAnalysisLanguage(settings: SettingsFormState): SettingsFormState {
  if (!settings.same_language_for_analysis) {
    return settings;
  }

  return {
    ...settings,
    analysis_language: settings.text_language,
  };
}

export function applyLoadedConfig(model: AppModel, config: PublicConfig): AppModel {
  return {
    ...model,
    config,
    settings: {
      ...model.settings,
      text_language: config.text_language,
      analysis_language: config.analysis_language,
      same_language_for_analysis: config.same_language_for_analysis,
      topic_prompt: config.last_topic_prompt,
      reuse_last_topic: false,
    },
    statusMessage: config.gemini.has_api_key
      ? "Stored API key is available."
      : "No API key stored yet. Placeholder generation still works in stage 3.",
  };
}

export function buildGenerateRequest(settings: SettingsFormState): {
  language: string;
  analysis_language: string;
  topic_prompt: string;
  reuse_last_topic: boolean;
} {
  return {
    language: settings.text_language,
    analysis_language: settings.analysis_language,
    topic_prompt: settings.topic_prompt.trim(),
    reuse_last_topic: settings.reuse_last_topic,
  };
}

export function buildConfigUpdatePayload(
  config: PublicConfig,
  settings: SettingsFormState,
): {
  text_language: string;
  analysis_language: string;
  same_language_for_analysis: boolean;
  ui_language: string;
  last_topic_prompt: string;
  session_limit: number;
  keep_last_audio: boolean;
  gemini: { model: string; api_key: string | null };
  langfuse: { host: string | null; public_key: string | null; secret_key: string | null };
} {
  return {
    text_language: settings.text_language,
    analysis_language: settings.analysis_language,
    same_language_for_analysis: settings.same_language_for_analysis,
    ui_language: config.ui_language,
    last_topic_prompt: settings.topic_prompt.trim(),
    session_limit: config.session_limit,
    keep_last_audio: config.keep_last_audio,
    gemini: {
      model: config.gemini.model,
      api_key: settings.api_key.trim() === "" ? null : settings.api_key.trim(),
    },
    langfuse: {
      host: config.langfuse.host,
      public_key: null,
      secret_key: null,
    },
  };
}

export function startGeneration(model: AppModel): AppModel {
  return {
    ...model,
    state: "generating_text",
    errorMessage: null,
    statusMessage: "Generating placeholder exercise...",
  };
}

export function applyGeneratedExercise(
  model: AppModel,
  exercise: GeneratedExercise,
): AppModel {
  return {
    ...model,
    state: "step_1_slow",
    exercise,
    errorMessage: null,
    statusMessage: "Exercise ready. Start with slow, clear reading.",
    settings: {
      ...model.settings,
      topic_prompt: exercise.topic_prompt,
      reuse_last_topic: false,
    },
  };
}

export function applyGenerationError(model: AppModel, message: string): AppModel {
  return {
    ...model,
    state: "error",
    errorMessage: message,
    statusMessage: null,
  };
}

export function advanceExerciseStep(model: AppModel): AppModel {
  if (model.state === "step_1_slow") {
    return {
      ...model,
      state: "step_2_natural",
      statusMessage: "Move to a more natural pace while keeping clarity.",
    };
  }

  if (model.state === "step_2_natural") {
    return {
      ...model,
      state: "step_3_retell_ready",
      statusMessage: "You are ready for the free retelling step.",
    };
  }

  return model;
}
```

```typescript
async function requestJson<T>(url: string, init?: RequestInit): Promise<T> {
  const response = await fetch(url, {
    headers: { "Content-Type": "application/json", ...(init?.headers ?? {}) },
    ...init,
  });

  if (!response.ok) {
    const detail = await response.text();
    throw new Error(detail || `${response.status} ${response.statusText}`);
  }

  return (await response.json()) as T;
}

function render(): void {
  const showExercise = model.state !== "home" && model.state !== "generating_text" && model.state !== "error";
  setupScreen.hidden = showExercise;
  exerciseScreen.hidden = !showExercise;
  settingsPanel.hidden = !model.settingsOpen;

  if (model.statusMessage) {
    statusBanner.hidden = false;
    statusBanner.textContent = model.statusMessage;
  } else if (model.errorMessage) {
    statusBanner.hidden = false;
    statusBanner.textContent = model.errorMessage;
  } else {
    statusBanner.hidden = true;
    statusBanner.textContent = "";
  }

  textLanguageInput.value = model.settings.text_language;
  analysisLanguageInput.value = model.settings.analysis_language;
  sameLanguageToggle.checked = model.settings.same_language_for_analysis;
  topicInput.value = model.settings.topic_prompt;
  apiKeyStatus.textContent = model.config?.gemini.has_api_key
    ? `Stored API key source: ${model.config.gemini.api_key_source}`

async function loadConfig(): Promise<void> {
  try {
    const config = await requestJson<PublicConfig>("/api/config", { method: "GET" });
    model = applyLoadedConfig(model, config);

async function generateExercise(): Promise<void> {
  if (!model.config) {
    model = applyGenerationError(model, "Config is not loaded yet.");
    render();
    return;
  }

  readSettingsFromInputs();
  model = startGeneration(model);
  render();

  try {
    const savedConfig = await requestJson<PublicConfig>("/api/config", {
      method: "POST",
      body: JSON.stringify(buildConfigUpdatePayload(model.config, model.settings)),
    });
    model = applyLoadedConfig(model, savedConfig);

    const exercise = await requestJson<GeneratedExercise>("/api/generate-text", {
      method: "POST",
      body: JSON.stringify(buildGenerateRequest(model.settings)),
    });
    model = applyGeneratedExercise(model, exercise);
  } catch {
    model = applyGenerationError(model, "Could not generate the placeholder exercise.");
  }

  render();
}
```
